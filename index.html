Easy Integration

Python SDK
from clisonix import ClisonixClient

# Initialize client
client = ClisonixClient()

# Login with credentials
login_data = client.login(
    "user@example.com",
    "password"
)
print(f"Token: {login_data['token']}")

# Use the API
health = client.health()
answer = client.ask("What is neural audio?")

# Auto-refresh if needed
client.refresh()

# Generate API key for production
api_key = client.create_api_key("prod-server")
TypeScript SDK
import { ClisonixClient } from './clisonix_sdk';

// Initialize client
const client = new ClisonixClient({
    baseUrl: 'https://api.clisonix.com'
});

// Login with email and password
const loginData = await client.login(
    'user@example.com',
    'password'
);
console.log(`Token: ${loginData.token}`);

// Make authenticated requests
const health = await client.health();
const answer = await client.ask("Hello!");

// Refresh token when needed
await client.refresh();

// Create API key for service
const apiKey = await client.createApiKey("backend-server");
